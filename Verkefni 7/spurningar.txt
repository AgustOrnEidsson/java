1. function greeting(name) {//þetta fall þarf að fá argument
	alert('Hello ' + name);//birtir Hello og svo argument með alert
   }

   function processUserInput(callback) {//þetta function er með argumenið callback sem verður fall
	var name = prompt('Please enter your name.');//gerir varið name
	callback(name);//hér tekur argumentið varið name og sendir í fallið sem callback er
   }

   processUserInput(greeting);//hér er kallað í fallið processUserInput og það tekur greeting sem argument
   //þetta verður þá þannig að greeting fallið verður keyrt innan í processUserInput
